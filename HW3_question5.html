import numpy as np
import matplotlib.pyplot as plt
from matplotlib.colors import ListedColormap
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.datasets import make_moons, make_circles, make_classification
from sklearn.neural_network import MLPClassifier
from sklearn.neighbors import KNeighborsClassifier
from sklearn.svm import SVC
from sklearn.gaussian_process import GaussianProcessClassifier
from sklearn.gaussian_process.kernels import RBF
from sklearn.tree import DecisionTreeClassifier
from sklearn.ensemble import RandomForestClassifier, AdaBoostClassifier
from sklearn.naive_bayes import GaussianNB
from sklearn.discriminant_analysis import QuadraticDiscriminantAnalysis

h = .02  # step size in the mesh

names = ["Nearest Neighbors", "Linear SVM", "RBF SVM",
         "Decision Tree", "Random Forest", "Naive Bayes"]
# linear SVC: penalty parameter C: large C, smaller margin hyperplane
# small C, larger magin hyperplane, may misclassify some points
# SVC kernel: rbf SVM: the gamma parameter defines how far the influence of a single training example reaches, 
# with low values meaning far and high values meaning close. 
# RandomForestClassifier: max_features: # of features to consider for the best split
classifiers = [
    KNeighborsClassifier(3),
    SVC(kernel="linear", C=0.025),
    SVC(gamma=2, C=1),
    DecisionTreeClassifier(max_depth=5),
    RandomForestClassifier(max_depth=5, n_estimators=10, max_features=1),
    GaussianNB()]

# generate a random n class classification problem
# random_state: Pass an int for reproducible output across multiple function calls
# n_classes = 2 by default
X, y = make_classification(n_features=2, n_redundant=0, n_informative=2,
                           random_state=1, n_clusters_per_class=1)

# container for random number generator, parameter is the seed
rng = np.random.RandomState(2)
X += 2 * rng.uniform(size=X.shape)
linearly_separable = (X, y)

# make_circles: a circle containing a small circle; scale factor: radius ratio
# all data sets have 2 features.Result of y: one of two possible classes.
datasets = [make_moons(noise=0.3, random_state=0),
            make_circles(noise=0.2, factor=0.5, random_state=1),
            linearly_separable
            ]

# figsize=(width, height)
figure = plt.figure(figsize=(27, 9))
i = 1
# iterate over datasets
for ds_cnt, ds in enumerate(datasets):
    # preprocess dataset, split into training and test part
    X, y = ds
    # StandardScaler: (x - mean) / standard deviation
    # fit and transform both X and y, but return new_X
    # fit only calculate mean and standard deviation of training set
    # when transform, apply the calculated mean and standard deviation to 
    # any set(training or testing or new examples)
    # ? How about y? y does not seem to be changed here.
    X = StandardScaler().fit_transform(X)
    X_train, X_test, y_train, y_test = \
        train_test_split(X, y, test_size=.4, random_state=42)

    # two features of X
    x_min, x_max = X[:, 0].min() - .5, X[:, 0].max() + .5
    y_min, y_max = X[:, 1].min() - .5, X[:, 1].max() + .5
    # xx, yy same shape, different elements
    xx, yy = np.meshgrid(np.arange(x_min, x_max, h),
                         np.arange(y_min, y_max, h))

    # just plot the dataset first
    cm = plt.cm.RdBu # two colors
    cm_bright = ListedColormap(['#FF0000', '#0000FF'])
    # subplot: rows, cols, index; index starts at 1 in the upper left corner, increases to the right
    ax = plt.subplot(len(datasets), len(classifiers) + 1, i)
    if ds_cnt == 0:
        ax.set_title("Input data")
    # Plot the training points
    # cmap is only used when c is an array of floats
    # c: a sequence of numbers to be mapped with cmap, mapping to color according to y: class label
    ax.scatter(X_train[:, 0], X_train[:, 1], c=y_train, cmap=cm_bright,
               edgecolors='k')
    # Plot the testing points
    ax.scatter(X_test[:, 0], X_test[:, 1], c=y_test, cmap=cm_bright, alpha=0.6,
               edgecolors='k')
    ax.set_xlim(xx.min(), xx.max())
    ax.set_ylim(yy.min(), yy.max())
    ax.set_xticks(())
    ax.set_yticks(())
    i += 1

    # iterate over classifiers
    for name, clf in zip(names, classifiers):
        ax = plt.subplot(len(datasets), len(classifiers) + 1, i)
        clf.fit(X_train, y_train)
        score = clf.score(X_test, y_test)

        # Plot the decision boundary. For that, we will assign a color to each
        # point in the mesh [x_min, x_max]x[y_min, y_max].
        # np.c_: one way of concatenation to the last axis
        # xx.ravel(): result is a 1d array with all the same elements
        # decision boundary: that's the reason to generate matrices of [x_min...x_max] * [y_min...y_max]
        # decision_function(X): X: samples * features
        # need to make np.c_[xx.ravel(), yy.ravel()] by yourself: [[x1, y1], [x2, y1], [x3, y1], ...[xn, y1]...[xn, yn]]
        if hasattr(clf, "decision_function"):
            # Z shape: samples * classes. samples means all points in meshgrid here.
            # SVM: samples * k*(k-1)/2 classes: distance of each point to hyperplane between two classes
            Z = clf.decision_function(np.c_[xx.ravel(), yy.ravel()])
        else:
            Z = clf.predict_proba(np.c_[xx.ravel(), yy.ravel()])[:, 1]

        # Put the result into a color plot
        # Z is the height in 3d, cmap maps height into colors
        Z = Z.reshape(xx.shape)
        ax.contourf(xx, yy, Z, cmap=cm, alpha=.8)

        # Plot the training points
        ax.scatter(X_train[:, 0], X_train[:, 1], c=y_train, cmap=cm_bright,
                   edgecolors='k')
        # Plot the testing points
        ax.scatter(X_test[:, 0], X_test[:, 1], c=y_test, cmap=cm_bright,
                   edgecolors='k', alpha=0.6)

        ax.set_xlim(xx.min(), xx.max())
        ax.set_ylim(yy.min(), yy.max())
        ax.set_xticks(())
        ax.set_yticks(())
        if ds_cnt == 0:
            ax.set_title(name)
        # ax.text(x, y, text): x, y is the position. score of test set.
        ax.text(xx.max() - .3, yy.min() + .3, ('%.2f' % score).lstrip('0'),
                size=15, horizontalalignment='right')
        i += 1

plt.tight_layout()
plt.show()

# RBF SVM always has a higher accuracy among algorithms. Decision trees works well on linearly separable data set.
# This data set doesn't show high correlation, so naive bayes works well as well.
